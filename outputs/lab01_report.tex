\subsection{AR Model Optimization Parameters}
\begin{lstlisting}[caption={Standard Least Squares predicting bounded matrices iteratively.}]
p_ar = 10 
N_samples = len(x1)
A = np.zeros((N_samples - p_ar, p_ar))
b_vec = x1[p_ar:]
for i in range(p_ar):
    A[:, i] = x1[p_ar - 1 - i : N_samples - 1 - i]
Theta = np.linalg.solve(A.T @ A, A.T @ b_vec)
a_ar = np.concatenate(([1], -Theta))
w_ar, h_ar = signal.freqz(1, a_ar, worN=1000, fs=Fs)
\end{lstlisting}
